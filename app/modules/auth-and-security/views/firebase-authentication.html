// Firebase API OAuth Login
firebase.auth()
.signInWithPopup('google')
.then(function(userCredential) {
  console.log(userCredential);
});

// AngularFire OAuth Login
$firebaseAuth()
.$signInWithPopup('google')
.then(function(user) {
  console.log(user);
});

// AngularFire Logout
$firebaseAuth().$signOut();
