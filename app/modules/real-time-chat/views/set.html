This will overwrite any data at this location and all child locations.

Returns: non-null firebase.Promise containing void Resolves when write to server is complete.ence; resolves when write is complete, but can be used immediately as the Reference to the child location.

var myData = {
  content: 'Hello, World!',
  timestamp: Date.now()
}

var db = firebase.database().ref();
var itemRef = db
  .child('/chats')
  .child(chatId)
.set(myData);

itemRef.then(function(snapshot) {
  console.log(snapshot);
});
